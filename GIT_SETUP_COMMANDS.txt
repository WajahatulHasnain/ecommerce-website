# Git Setup Commands - Run these in PowerShell

# 1. Initialize git (if not already done)
git init

# 2. Add all files
git add .

# 3. Commit
git commit -m "Initial commit - SnapShop E-commerce"

# 4. Create GitHub repo at: https://github.com/new
# Name: snapshop-ecommerce
# Keep it PUBLIC
# Do NOT initialize with README

# 5. Add GitHub remote (replace YOUR_USERNAME)
git remote add origin https://github.com/YOUR_USERNAME/snapshop-ecommerce.git

# 6. Push to GitHub
git branch -M main
git push -u origin main
